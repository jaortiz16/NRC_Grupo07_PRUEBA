JQuery Traversing Ancestors

Ejemplo de uso de la función parent() de jQuery



Ubicacion del texto ingresado






$(document).ready(function() {
      $('#parentButton').click(function() {
        var inputText = $('#inputField').val();
        var parentElement = $('#inputField').parent();
        var parentId = parentElement.attr('id');
        
        $('#inputField').addClass('red-text');
        alert('Texto ingresado: ' + inputText + '\nID del padre: ' + parentId);

        $('#inputField').removeClass('red-text').addClass('green-bold').val(inputText.toUpperCase());
        
        var resultContainer = $('#resultContainer');
        resultContainer.text(inputText.toUpperCase());
        resultContainer.width(resultContainer.textWidth() + 15);
      });
    });

    // Extensión de jQuery para obtener el ancho del texto
    $.fn.textWidth = function() {
      var html_org = $(this).html();
      var html_calc = '' + html_org + '';
      $(this).html(html_calc);
      var width = $(this).find('span:first').width();
      $(this).html(html_org);
      return width;
    };

Ejemplo de la funcion parents()

$(document).ready(function() {
      $('.celeste-button').click(function() {
        var parents = $('.inner-div').parents();
        var message = 'IDs de los elementos padres:\n';

        parents.each(function() {
          var parentId = $(this).attr('id');
          message += parentId + '\n';
        });

        alert(message);
      });
    });

Ejemplo parentsUntil()

$(document).ready(function() {
      $('.celeste-button').click(function() {
        var parents = $('.inner-div').parentsUntil('.cuerpo');
