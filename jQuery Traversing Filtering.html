    $(document).ready(function() {
      $('#filtrar-btn').click(function() {
        var clase = $('#clase-input').val();
        $('#deportes-list li').removeClass('filter-resaltado');
        var elementosFiltrados = $('#deportes-list li').filter('.' + clase);
        elementosFiltrados.addClass('filter-resaltado');
        mostrarResultado(elementosFiltrados);
      });

      function mostrarResultado(elementos) {
        var resultado = $('#resultado-list');
        resultado.empty();
        elementos.each(function() {
          resultado.append('
  • ' + $(this).text() + '
    • ');
        });
      }
    });
