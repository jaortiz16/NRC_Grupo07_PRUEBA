JQuery Traversing Descendants

Ejemplo Children()

$(document).ready(function() {
      $('.button').click(function() {
        var childrenIDs = [];
        getChildrenIDs($('.container'), childrenIDs);
        
        alert(childrenIDs.join(', '));
      });
    });
    
    function getChildrenIDs(element, childrenIDs) {
      element.children().each(function() {
        childrenIDs.push($(this).attr('id'));
        if ($(this).children().length > 0) {
          getChildrenIDs($(this), childrenIDs);
        }
      });
    }

$(document).ready(function() {

        var result = $(".container").find(".inner-div img");
        if (result.length > 0) {
          var elementId = result.closest(".container").attr("id");
          alert("Se encontró una imagen en el elemento con ID: " + elementId);
        } else {
          alert("No se encontró ninguna imagen en el contenedor.");
        }
