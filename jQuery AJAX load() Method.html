Teoria de programacion

Cambio de texto con JQUERY AJAX 

$(document).ready(function(){
  $("button").click(function(){
    $.get("programacion.txt", function(data){
      $("#div1").html("¡Contenido externo cargado exitosamente!");
      alert("¡Contenido externo cargado exitosamente!");
    }).fail(function(xhr, status, error) {
      $("#div1").html("Error: " + xhr.status + ": " + error);
      alert("Error: " + xhr.status + ": " + error);
    });
  });
});

Resumen de pelicula

Texto cambiado por jQuery AJAX 

$(document).ready(function(){
  $("button").click(function(){
    $("#div1").load("peliculas.txt", function(responseTxt, statusTxt, xhr){
      if(statusTxt == "success")
        $("#div1").html("¡Contenido externo cargado exitosamente!");
      if(statusTxt == "error")
        $("#div1").html("Error el contenido no se pudo cargar : " + xhr.status + ": " + xhr.statusText);
    });
  });
});
