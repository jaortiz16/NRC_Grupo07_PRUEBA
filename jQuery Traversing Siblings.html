Selecciona de jugadores de futbol

$(document).ready(function() {
      $("#resaltar-jugador1").click(function() {
        $(".jugador1:nth-child(3)").siblings().removeClass("resaltado1");
        $(".jugador1:nth-child(3)").addClass("resaltado1");
      });

      $("#resaltar-jugador2").click(function() {
        $(".jugador1:nth-child(4)").siblings().removeClass("resaltado1");
        $(".jugador1:nth-child(4)").addClass("resaltado1");
      });

      $("#resaltar-jugador3").click(function() {
        $(".jugador1:nth-child(5)").siblings().removeClass("resaltado1");
        $(".jugador1:nth-child(5)").addClass("resaltado1");
      });
      
      $("#resaltar-jugador4").click(function() {
        $(".jugador1:nth-child(6)").siblings().removeClass("resaltado1");
        $(".jugador1:nth-child(6)").addClass("resaltado1");
      });
      
      $("#resaltar-jugador5").click(function() {
        $(".jugador1:nth-child(7)").siblings().removeClass("resaltado1");
        $(".jugador1:nth-child(7)").addClass("resaltado1");
      });
    });
